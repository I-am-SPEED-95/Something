CREATE TABLE order_details (
    ORDER_CODE               INTEGER             NOT NULL,
    PRODUCT_NUMBER           INT             NOT NULL,
    RETAILER_NAME               VARCHAR(100),
    CUSTOMER_STORE_SK           INT             NOT NULL,
    CUSTOMER_CONTACT_SK         INT             NOT NULL,
    SALES_REPRESENTATIVE_SK  INT             NOT NULL,
    SALES_BRANCH_SK          INT             NOT NULL,
    ORDER_DATE                  DATE,
    ORDER_DAY                   INT,
    ORDER_MONTH                 INT,
    ORDER_YEAR                  INT,
    ORDER_METHOD_CODE        INT             NOT NULL,
    MARGIN                      DECIMAL(15,2),
    DISCOUNT                    DECIMAL(10,4),
    TURNOVER                    DECIMAL(15,2),
    PROFIT                      DECIMAL(15,2),
    QUANTITY                    INT,
    UNIT_COST                   DECIMAL(15,2),
    UNIT_SALE_PRICE             DECIMAL(15,2),
    UNIT_PRICE                  DECIMAL(15,2),
 
CONSTRAINT PK_ORDER PRIMARY KEY (ORDER_CODE),
CONSTRAINT FK_ORDER_PRODUCT     FOREIGN KEY (PRODUCT_NUMBER) REFERENCES product (PRODUCT_NUMBER),
CONSTRAINT FK_ORDER_CUSTOMER    FOREIGN KEY (CUSTOMER_STORE_SK) REFERENCES customer_site (CUSTOMER_STORE_SK),
CONSTRAINT FK_ORDER_CONTACT     FOREIGN KEY (CUSTOMER_CONTACT_SK) REFERENCES Customer_Contact (CUSTOMER_CONTACT_SK),
CONSTRAINT FK_ORDER_SALES_REP   FOREIGN KEY (SALES_REPRESENTATIVE_SK) REFERENCES sales_representative (SALES_REPRESENTATIVE_SK),
CONSTRAINT FK_ORDER_BRANCH      FOREIGN KEY (SALES_BRANCH_SK) REFERENCES sales_branch (SALES_BRANCH_SK),
CONSTRAINT FK_ORDER_METHOD      FOREIGN KEY (ORDER_METHOD_CODE) REFERENCES order_method (ORDER_METHOD_CODE),
CONSTRAINT FK_DATE              FOREIGN KEY (ORDER_DAY, ORDER_MONTH, ORDER_YEAR) REFERENCES date (DAY, MONTH, YEAR)
);
